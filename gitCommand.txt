
git init  راه اندازی گیت 

git clone [url]  کپی گرفتن از ریپازیتوری در لوکال خودمون

git config -global user.name "[اسم]"

git config -global user.email "[ایمیل]"

git add  اضافه کردن فایل مورد نظر   [اسم فایل ]   

git add .  اضافه کردن همه ی فایل ها در مرحله استیج

git commit -m "[message]"  فایل هایی که در استیج وجود دارند رو کامیت میکنه و در جایی ذخیره و قابل دسترس میزاره

git commit -a -m "[message]"  کار اد و کامیت رو باهم انجام میده ، ولی یه سری استثا هم داره 

git status  نمایش تغییرات 

git log مشاهده همه لاگ های موجود (کامیت ها)

git log --oneline مشاهده همه لاگ ها ولی بصورت خلاصه وار 

git revert [commit token] برگشت به کامیت مورد نظر 

git diff مشاهده دقیق تغییرات و تفاوت ها

git checkout کارهای زیادی میکنه ولی برای الان ، این دستور برای جا به جا شدن بین شاخه ها استفاده می شود

git branch با این دستور میتونیم شاخه های موجودی که داریم رو ببینم و نمایش بدیم 

git push  انتقال تمام فایل ها از لوکال به ریموت .

git pull گرفتن همه فایل ها از ریموت به انتقال اونها به لوکال خودمون
